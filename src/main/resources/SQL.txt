CREATE TABLE message_tasks (
    task_id INT PRIMARY KEY NOT NULL UNIQUE,
    task_name VARCHAR(255),
    delivery_channel VARCHAR(255),
    sender_recipient JSON,
    receiver_query VARCHAR(255),
    receiver_information JSON,
    template_id INT,
    time_type VARCHAR(50),
    send_time DATETIME,
    actual_send_time DATETIME,
    status VARCHAR(50),
    created_at DATETIME,
    updated_at DATETIME,
    ALIVE VARCHAR(50) NOT NULL DEFAULT 'YES'
);

CREATE TABLE events (
    event_id INT PRIMARY KEY NOT NULL UNIQUE,
    event_type VARCHAR(50) NOT NULL,
    event_data JSON NOT NULL,
    trigger_time DATETIME NOT NULL,
    task_id INT,
    created_at DATETIME NOT NULL,
    updated_at DATETIME,
    FOREIGN KEY (task_id) REFERENCES message_tasks(imessaged)
);
